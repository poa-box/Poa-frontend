# Allow search crawlers full access. App pages that should not appear in search
# carry a per-page <meta name="robots" content="noindex, nofollow"> via the
# SEOHead component — Google must be allowed to crawl the page in order to
# read that directive and properly de-index it. Listing those same paths under
# Disallow here would block crawling, which causes the "Indexed, though blocked
# by robots.txt" warning in Search Console (URL is indexed without content
# instead of being de-indexed).
User-agent: *
Allow: /

Disallow: /
